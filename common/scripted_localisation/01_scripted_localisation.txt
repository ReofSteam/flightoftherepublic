defined_text = {
	name = SUBIDEOLOGIES_OPEN
	text = {
		trigger = {
			NOT = {
				has_country_flag = subideologies_hasideology
			}
		}
		localization_key = "subideologies_open"
	}
	text = {
		trigger = {
			has_country_flag = subideologies_hasideology
		}
		localization_key = "subideologies_close"
	}
}
